```javascript
// example myModule.
directive(‘directiveName’, 
  function () { 
    return { 
      restrict: ‘E’,
      scope: { myValue: ‘=’}, 
      link: function(scope, el, attrs) { 
        ... 
      } 
    }; 
  });
```
